# 深入理解 javascirpt 系列

整理自汤姆大叔的《深入理解 javascirpt 系列》 (opens new window)，只有核心知识章节做了笔记

# 第一章 编写高质量 JavaScript 代码的基本要点

  1. 最少全局变量

  2. 忘记 var 的副作用

    • a.通过 var 创建的全局变量（任何函数之外的程序中创建）是不能被删除的

    • b.无 var 创建的隐式全局变量（无视是否在函数中创建）是能被删除的

  3. 单 var 形式声明

  4. 循环中的数组长度缓存

  5. 循环中通过hasOwnProperty避免原型属性

  6. 尽量不拓展内置原型

  7. 避免 switch 贯穿（每个 case 语句用 break 结束）

  8. 避免隐式类型转换，始终使用===而非==

  9. 避免eval()

  10. 避免在定时器函数调用中传递参数

  11. 单行代码中的 if 花括号不应该被省略

# 第二章 揭秘命名函数表达式

  1. 不能将函数声明写在 Block(块)语句中 if、while 或 for 语句,存在游览器差异

  2. 带函数名的函数表达式的名字只在其作用域能被访问到，在外围访问不到

  3. ()符为分组操作符，在其中的函数只能为表达式形式(函数表达式)

# 第三章 全面解析Module模式

# 独立模块

var blogModule = (function () {
    var my = {}, privateName = "博客园";
    function privateAddTopic(data) {//私有部分
        // 这里是内部处理代码
    }
    my.Name = privateName;	    
    my.AddTopic = function (data) {//公共部分
        privateAddTopic(data);
    };
    return my;
} ());

# 拓展模块

var blogModule = (function (my) {
    my.AddPhoto = function () {
        //添加内部代码  
    };
    return my;
} (blogModule || {}));

# 克隆与继承

var blogModule = (function (old) {
    var my = {},
        key;
    for (key in old) {
        if (old.hasOwnProperty(key)) {
            my[key] = old[key];
        }
    }
    var oldAddPhotoMethod = old.AddPhoto;
    my.AddPhoto = function () {
        // 克隆以后，进行了重写，当然也可以继续调用oldAddPhotoMethod
    };
    return my;
} (blogModule));

# 模块互访（跨文件共享私有对象）

var blogModule = (function (my) {
    var _private = my._private = my._private || {},
        _seal = my._seal = my._seal || function () {
            delete my._private;
            delete my._seal;
            delete my._unseal;
        },
        _unseal = my._unseal = my._unseal || function () {
            my._private = _private;
            my._seal = _seal;
            my._unseal = _unseal;
        };      
    return my;
} (blogModule || {}));

# 第四章 立即调用的函数表达式

# 自执行的函数表达式

  • 函数声明语句

    function a() {}

  • 函数表达式

    // 方式1（此时在后面加分组操作符()，会被理解为立即执行）
(function a(){})

// 方式2
var a=function(){}

括弧()和&&，异或，逗号等操作符是在函数表达式和函数声明上消除歧义的，一旦解析器知道其中一个已经是表达式了，其它的也都默认为表达式了，比如：

var i = function () { return 10; } ();
true && function () { /* code */ } ();
0, function () { /* code */ } ();
new function () { /* code */ } ();

# 自执行匿名函数和立即执行的函数表达式区别

// 自执行函数（函数内部执行自身，递归）
function foo() { foo(); }

// 自执行匿名函数
(function() { arguments.callee(); })

// 立即调用的函数表达式
var foo = function() {} ()

// 立即调用的函数表达式（立即执行的匿名函数）
(function(){})()

# 未声明的函数语句和声明的函数语句的区别

// 未声明的函数语句（执行报错）
// 原因：js解析function关键字的时候，默认为function声明，而此时刚好没名字
function(){}()

// 声明的函数语句（执行报错）
// 原因：分组操作符()括号内无表达式
function a(){}()

// 声明的函数语句（不报错，但无意义）
// 原因：该声明函数作为函数语句，其后跟分组操作符带"1"的表达式，不是在执行该函数，而是属于两句独立的代码语句，各自执行
function a(){}(1)

// 声明的函数表达式（执行成功）
// 原因：第一个分组操作符将本该被js解析器识别的声明函数变成了声明的函数表达式，其后的分组操作符()被作为了函数执行
(function a(){})()
